## Install

```bash
npm install --save react-wallet-selector
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-wallet-selector'
import 'react-wallet-selector/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
